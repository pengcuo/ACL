117 __kernel void reduction_operation_x(
118  IMAGE_DECLARATION(src),
119  IMAGE_DECLARATION(partial_res),
120  __local DATA_TYPE *local_results)
121 {
122  Image src = CONVERT_TO_IMAGE_STRUCT(src);
123  Image partial_res = CONVERT_TO_IMAGE_STRUCT(partial_res);
124 
125  unsigned int lsize = get_local_size(0);
126  unsigned int lid = get_local_id(0);
127 
128  for(unsigned int y = 0; y < get_local_size(1); ++y)
129  {
130  local_results[lid] = OPERATION((__global DATA_TYPE *)offset(&src, 0, y));
131  barrier(CLK_LOCAL_MEM_FENCE);
132 
133  // Perform parallel reduction
134  for(unsigned int i = lsize >> 1; i > 0; i >>= 1)
135  {
136  if(lid < i)
137  {
138 #if defined(PROD)
139  local_results[lid] *= local_results[lid + i];
140 #else // !defined(PROD)
141  local_results[lid] += local_results[lid + i];
142 #endif // defined(PROD)
143  }
144  barrier(CLK_LOCAL_MEM_FENCE);
145  }
146 
147  if(lid == 0)
148  {
149 #if defined(MEAN) && defined(WIDTH)
150  if(y == get_local_size(1) - 1)
151  {
152  local_results[0] /= WIDTH;
153  }
154 #endif // defined(MEAN) && defined(WIDTH)
155  ((__global DATA_TYPE *)offset(&partial_res, get_group_id(0), y))[0] = local_results[0];
156  }
157  }
158 }
159 #endif // defined(OPERATION)

178 __kernel void reduction_operation_non_parallel_x(
179  VECTOR_DECLARATION(src),
180  VECTOR_DECLARATION(output))
181 {
182  Vector src = CONVERT_TO_VECTOR_STRUCT(src);
183  Vector output = CONVERT_TO_VECTOR_STRUCT(output);
184 
185  DATA_TYPE_PROMOTED res = *((__global DATA_TYPE *)vector_offset(&src, 0));
186 
187 #if defined(ARG_MAX) || defined(ARG_MIN)
188  uint indx = 0;
189 #endif // defined(ARG_MAX) || defined(ARG_MIN)
190 
191  for(unsigned int x = 1; x < WIDTH; ++x)
192  {
193  DATA_TYPE_PROMOTED in = *((__global DATA_TYPE *)vector_offset(&src, x));
194 #if defined(ARG_MAX)
195  indx = select(indx, x, ISGREATER(in, res));
196  res = select(res, in, CONVERT(ISGREATER(in, res), COND_DATA_TYPE));
197 #elif defined(ARG_MIN)
198  indx = select(indx, x, ISLESS(in, res));
199  res = select(res, in, CONVERT(ISLESS(in, res), COND_DATA_TYPE));
200 #elif defined(MIN)
201  res = select(res, in, CONVERT(ISLESS(in, res), COND_DATA_TYPE));
202 #elif defined(MAX)
203  res = select(res, in, CONVERT(ISGREATER(in, res), COND_DATA_TYPE));
204 #else // !(defined(ARG_MAX) || defined(ARG_MIN))
205  res += in;
206 #endif // defined(ARG_MAX) || defined(ARG_MIN)
207  }
208 
209  // Store result
210 #if defined(ARG_MAX) || defined(ARG_MIN)
211  *((__global uint *)output.ptr) = indx;
212 #else // !(defined(ARG_MAX) || defined(ARG_MIN))
213 #if defined(MEAN)
214  res /= WIDTH;
215 #endif // defined(MEAN)
216 #if defined(MIN) || defined(MAX)
217  *((__global DATA_TYPE_PROMOTED *)output.ptr) = res;
218 #else // defined(MIN) || defined(MAX)
219  *((__global uchar *)output.ptr) = convert_uchar(res);
220 #endif // defined(MIN) || defined(MAX)
221 #endif // defined(ARG_MAX) || defined(ARG_MIN)
222 }
223 #endif // defined(WIDTH)
