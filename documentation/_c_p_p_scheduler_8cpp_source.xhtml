80 void process_workloads(std::vector<IScheduler::Workload> &workloads, ThreadFeeder &feeder, const ThreadInfo &info)
81 {
82  unsigned int workload_index = info.thread_id;
83  do
84  {
85  ARM_COMPUTE_ERROR_ON(workload_index >= workloads.size());
86  workloads[workload_index](info);
87  }
88  while(feeder.get_next(workload_index));
89 }

177 void CPPScheduler::Thread::worker_thread()
178 {
179  while(true)
180  {
181  std::unique_lock<std::mutex> lock(_m);
182  _cv.wait(lock, [&] { return _wait_for_work; });
183  _wait_for_work = false;
184 
185  _current_exception = nullptr;
186 
187  // Time to exit
188  if(_workloads == nullptr)
189  {
190  return;
191  }
192 
193 #ifndef ARM_COMPUTE_EXCEPTIONS_DISABLED
194  try
195  {
196 #endif /* ARM_COMPUTE_EXCEPTIONS_ENABLED */
197  process_workloads(*_workloads, *_feeder, _info);
198 
199 #ifndef ARM_COMPUTE_EXCEPTIONS_DISABLED
200  }
201  catch(...)
202  {
203  _current_exception = std::current_exception();
204  }
205 #endif /* ARM_COMPUTE_EXCEPTIONS_DISABLED */
206  _job_complete = true;
207  lock.unlock();
208  _cv.notify_one();
209  }
210 }

237 {

239  if(num_threads < 1)
240  {
241  return;
242  }
243  ThreadFeeder feeder(num_threads, workloads.size());
244  ThreadInfo info;
245  info.cpu_info = &_cpu_info;
246  info.num_threads = num_threads;

248  auto thread_it = _threads.begin();
249  for(; t < num_threads - 1; ++t, ++thread_it)
250  {
251  info.thread_id = t;
252  thread_it->start(&workloads, feeder, info);
253  }
254 
255  info.thread_id = t;
256  process_workloads(workloads, feeder, info);
257 #ifndef ARM_COMPUTE_EXCEPTIONS_DISABLED
258  try
259  {
260 #endif /* ARM_COMPUTE_EXCEPTIONS_DISABLED */

275 void CPPScheduler::schedule(ICPPKernel *kernel, const Hints &hints)
276 {
277  ARM_COMPUTE_ERROR_ON_MSG(!kernel, "The child class didn't set the kernel");
278 
279  const Window &max_window = kernel->window();
280  const unsigned int num_iterations = max_window.num_iterations(hints.split_dimension());
281  const unsigned int num_threads = std::min(num_iterations, _num_threads);
282 
283  if(num_iterations == 0)
284  {
285  return;
286  }
287 
288  if(!kernel->is_parallelisable() || num_threads == 1)
289  {
290  ThreadInfo info;
291  info.cpu_info = &_cpu_info;
292  kernel->run(max_window, info);
293  }
294  else
295  {
296  unsigned int num_windows = 0;
297  switch(hints.strategy())
298  {
299  case StrategyHint::STATIC:
300  num_windows = num_threads;
301  break;
302  case StrategyHint::DYNAMIC:
303  {
304  // Make sure we don't use some windows which are too small as this might create some contention on the ThreadFeeder
305  const unsigned int max_iterations = static_cast<unsigned int>(_num_threads) * 3;
306  num_windows = num_iterations > max_iterations ? max_iterations : num_iterations;
307  break;
308  }
309  default:
310  ARM_COMPUTE_ERROR("Unknown strategy");
311  }
312  std::vector<IScheduler::Workload> workloads(num_windows);
313  for(unsigned int t = 0; t < num_windows; t++)
314  {
315  //Capture 't' by copy, all the other variables by reference:
316  workloads[t] = [t, &hints, &max_window, &num_windows, &kernel](const ThreadInfo & info)
317  {
318  Window win = max_window.split_window(hints.split_dimension(), t, num_windows);
319  win.validate();
320  kernel->run(win, info);
321  };
322  }
323  run_workloads(workloads);
324  }
325 }
