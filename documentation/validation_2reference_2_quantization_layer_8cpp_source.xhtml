37 SimpleTensor<uint8_t> quantization_layer(const SimpleTensor<T> &src, const QuantizationInfo &quantization_info)
38 {
39  // Create reference
40  SimpleTensor<uint8_t> dst{ src.shape(), DataType::QASYMM8, 1, quantization_info };
41 
42  const UniformQuantizationInfo qinfo = quantization_info.uniform();
43  for(int i = 0; i < src.num_elements(); ++i)
44  {
45 #ifdef __aarch64__
46  dst[i] = quantize_qasymm8((src[i]), qinfo, RoundingPolicy::TO_NEAREST_EVEN);
47 #else // __aarch64__
48  dst[i] = quantize_qasymm8((src[i]), qinfo, RoundingPolicy::TO_ZERO);
49 #endif // __aarch64__
50  }
51  return dst;
52 }
